simple ms procedure to build the client I couldn't get webpack to work here so I'm doing that externally.

 
Metalsmith(baseDir)
.source(resolve(baseDir, 'lib', 'client'))
.destination(resolve(baseDir, 'dist'))
.use(webpack())
.use(pug({ useMetadata: true }))
.use(ignore([

'styles.less',

 
  '**/*.jsx',
  '**/*.less',
  '**/*.pug'
]))

.build((err) => {
  if (err) dbg(err)
  else dbg('done')
})
